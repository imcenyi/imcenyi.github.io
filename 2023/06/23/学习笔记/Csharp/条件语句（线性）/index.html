    void Func1()

    {

        int value1=15;

        if(value1>5)

        {

            Debug.Log("value1 is greater than 5");//value1满足了第一个条件就执行了，输出语句

        }

        else if(value1>10)

        {

            Debug.Log("value1 is greater than 10");//由于第一个已经执行了，就终端这一串条件语句，所以虽然满足条件但没有执行

        }

        else

        {

            Debug.Log("value1 is less than 5");

        }

    }

}
